1use core::sync::atomic::Ordering::*;
2use core::sync::atomic::{AtomicBool, AtomicPtr};
3
4use crossbeam_utils::{Backoff, CachePadded};
5
6use crate::lock::*;
7
8struct Node {
9    locked: AtomicBool,
10}
11
12#[derive(Debug, Clone)]
13pub struct Token(*const CachePadded<Node>);
14
15// SAFETY: It doesn't matter if a thread used a token made by another thread.
16unsafe impl Send for Token {}
17
18/// CLH lock.
19#[derive(Debug)]
20pub struct ClhLock {
21    tail: AtomicPtr<CachePadded<Node>>,
22}
23
24impl Node {
25    fn new(locked: bool) -> *mut CachePadded<Self> {
26        Box::into_raw(Box::new(CachePadded::new(Self {
27            locked: AtomicBool::new(locked),
28        })))
29    }
30}
31
32impl Default for ClhLock {
33    fn default() -> Self {
34        let node = AtomicPtr::new(Node::new(false));
35
36        Self { tail: node }
37    }
38}
39
40unsafe impl RawLock for ClhLock {
41    type Token = Token;
42
43    fn lock(&self) -> Self::Token {
44        let node = Node::new(true);
45        let prev = self.tail.swap(node, AcqRel);
46        let backoff = Backoff::new();
47
48        // SAFETY: `prev` is valid, as `self.tail` was valid at initialization and any `swap()` to
49        // it by other `lock()`s. Hence, it points to valid memory as the thread that made `prev`
50        // will not free it.
51        while unsafe { (*prev).locked.load(Acquire) } {
52            backoff.snooze();
53        }
54
55        // SAFETY: since `prev` was obtained from a swap on tail, only this thread other than its
56        // creator can access it. Since the creator will no longer access `prev` as its `locked` is
57        // false, we have unique access to it.
58        drop(unsafe { Box::from_raw(prev) });
59        Token(node)
60    }
61
62    unsafe fn unlock(&self, token: Self::Token) {
63        unsafe { (*token.0).locked.store(false, Release) };
64    }
65}
66
67impl Drop for ClhLock {
68    fn drop(&mut self) {
69        // Drop the node made by the last thread that `lock()`ed.
70        let node = *self.tail.get_mut();
71
72        // SAFETY: Since this is the tail node, no other thread has access to it.
73        drop(unsafe { Box::from_raw(node) });
74    }
75}
